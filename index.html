¿Cuántos migrantes venezolanos han entrado a Colombia desde 2012 hasta 2022?

SELECT A__o, SUM(total) AS total_sumado
FROM `migrantes-venezolanos-a-col.entrada.datos`
GROUP BY A__o
ORDER BY A__o
